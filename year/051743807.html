单调栈

洛谷P1950 长方形

单调栈可以在O(n)的时间内求出一个数左/右边比他小/大的第一个数

#include <iostream>
#include <stack>
using namespace std;
const int N = 1005;
int n, m;
int a[N][N];
int l[N], r[N], h[N];
long long ans;
void workl() {
    stack<int> stk;
    for (int i = m; i >= 1; i--) {
        while (stk.size() && h[i] <= h[stk.top()]) {
            l[stk.top()] = i;
            stk.pop();
        }
        stk.push(i);
    }
    while (stk.size()) {
        l[stk.top()] = 0;
        stk.pop();
    }
}
void workr() {
    stack<int> stk;
    for (int i = 1; i <= m; i++) {
        while (stk.size() && h[i] < h[stk.top()]) {
            r[stk.top()] = i;
            stk.pop();
        }
        stk.push(i);
    }
    while (stk.size()) {
        r[stk.top()] = m + 1;
        stk.pop();
    }
}
void work() {
    workl();
    workr();
    for (int i = 1; i <= m; i++) {
        ans += (i - l[i]) * (r[i] - i) * h[i];
    }
}
int main() {
    char c;
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cin >> c;
            if (c == '*') a[i][j] = 1;
        }
    }
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            h[j]++;
            if (a[i][j]) h[j] = 0;
        }
        work();
    }
    printf("%lld\n", ans);
    return 0;
}

const int N = 60;
int pre[N], in[N];
int v[N], lc[N], rc[N], idx;
int reduce(int ps, int is, int ie) {
    int u = pre[ps];
    v[++idx] = u;
    int id = idx, k = 0;
    for (int i = is; i <= ie; i++) {
        if (in[i] == u) break;
        k++;
    }
    if (k == ie - is + 1) return 0;
    if (k != 0) lc[id] = reduce(ps + 1, is, is + k - 1);
    if (k != ie - is) rc[id] = reduce(ps + k + 1, is + k + 1, ie);
    return id;
}

#include <iostream>

const int N = 1e5 + 5, M = 31 * N;
int son[M][2], idx;
void insert(int x) {
    int p = 0; 
    for (int i = 30; i >= 0; i--) {
        int u = x >> i & 1;
        if (!son[p][u]) son[p][u] = ++idx;
        p = son[p][u];
    }
}
int search(int x) {
    int p = 0, res = 0;
    for (int i = 30; i >= 0; i--) {
        int u = x >> i & 1;
        if (son[p][!u]) {
            res = res * 2 + !u;
            p = son[p][!u];
        } else {
            res = res * 2 + u;
            p = son[p][u];
        }
    }
    return res;
}
int main() {
    int n;
    cin >> n;
    int res = 0;
    for (int i = 0; i < n; i++) {
        int a;
        cin >> a;
        insert(a);
        int t = search(a);
        res = max(res, a ^ t);
    }

}

const int N = 100010;
int h[N], ph[N], hp[N], idx;
void heap_swap(int a, int b) {
    swap(h[a], h[b]);
    swap(ph[hp[a]], ph[hp[b]]);
    swap(hp[a], hp[b]);
}
void down(int u) {
    int t = u;
    if (u * 2 <= idx && h[u * 2] < h[t]) t = u * 2;
    if (u * 2 + 1 <= idx && h[u * 2 + 1] < h[t]) t = u * 2 + 1;
    if (u != t) {
        heap_swap(u, t);
        down(t);
    }
} 
void up(int u) {
    while (u / 2 && h[u] < h[u / 2]) {
        heap_swap(u, u / 2);
        u /= 2;
    }
}

#include <iostream>
using namespace std;
#define int long long
const int N = 1e5 + 5;
struct Node {
    int l, r;
    int sum;
    int add;
}tr[N * 4];
int a[N];
void pushup(int u) {
    tr[u].sum = tr[u << 1].sum + tr[u << 1 | 1].sum;
}
void pushdown(int u) {
    Node &rt = tr[u], &lc = tr[u << 1], &rc = tr[u << 1 | 1];
    if (rt.add == 0) return;
    lc.add += rt.add;
    lc.sum += (lc.r - lc.l + 1) * rt.add;
    rc.add += rt.add;
    rc.sum += (rc.r - rc.l + 1) * rt.add;
    rt.add = 0;
}
void build(int u, int l, int r) {
    if (l == r) tr[u] = {l, r, a[r], 0};
    else {
        tr[u] = {l, r};
        int mid = l + r >> 1;
        build(u << 1, l, mid);
        build(u << 1 | 1, mid + 1, r);
        pushup(u);
    }
}
void modify(int u, int l, int r, int k) {
    if (tr[u].l > r || tr[u].r < l) return;
    else if (tr[u].l >= l && tr[u].r <= r) {
        tr[u].add += k;
        tr[u].sum += (tr[u].r - tr[u].l + 1) * k;
    } else {
        pushdown(u);
        modify(u << 1, l, r, k);
        modify(u << 1 | 1, l, r, k);
        pushup(u);
    }
}
int query_sum(int u, int l, int r) {
    if (tr[u].l > r || tr[u].r < l) return 0;
    if (tr[u].l >= l && tr[u].r <= r) return tr[u].sum;
    pushdown(u);
    return query_sum(u << 1, l, r) + query_sum(u << 1 | 1, l, r);
}
signed main() {
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i++) cin >> a[i];
    build(1, 1, n);
    for (int i = 0; i < m; i++) {
        int op, l, r;
        cin >> op >> l >> r;
        if (op == 1) {
            int k;
            cin >> k;
            modify(1, l, r, k);
        } else {
            cout << query_sum(1, l, r) << endl;

}

#include <iostream>
using namespace std;
#define int long long
const int N = 1e5 + 5;
struct Node {
    int l, r;
    int sum;
    int add, mul;
}tr[N * 4];
int a[N];
int p;
void pushup(int u) {
    tr[u].sum = (tr[u << 1].sum + tr[u << 1 | 1].sum) % p;
}
void update(int u, int add, int mul) {
    Node &rt = tr[u];
    rt.sum = (rt.sum * mul + (rt.r - rt.l + 1) * add) % p;
    rt.add = (rt.add * mul + add) % p;
    rt.mul = rt.mul * mul % p;
}
void pushdown(int u) {
    Node &rt = tr[u];
    if (rt.add == 0 && rt.mul == 1) return;
    update(u << 1, rt.add, rt.mul);
    update(u << 1 | 1, rt.add, rt.mul);
    rt.add = 0, rt.mul = 1;
}
void build(int u, int l, int r) {
    if (l == r) tr[u] = {l, r, a[r], 0, 1};
    else {
        tr[u] = {l, r, 0, 0, 1};
        int mid = l + r >> 1;
        build(u << 1, l, mid);
        build(u << 1 | 1, mid + 1, r);
        pushup(u);
    }
}
void modify(int u, int l, int r, int add, int mul) {
    if (tr[u].l > r || tr[u].r < l) return;
    else if (tr[u].l >= l && tr[u].r <= r) {
        update(u, add, mul);
    } else {
        pushdown(u);
        modify(u << 1, l, r, add, mul);
        modify(u << 1 | 1, l, r, add, mul);
        pushup(u);
    }
}
int query_sum(int u, int l, int r) {
    if (tr[u].l > r || tr[u].r < l) return 0;
    if (tr[u].l >= l && tr[u].r <= r) return tr[u].sum;
    pushdown(u);
    return (query_sum(u << 1, l, r) + query_sum(u << 1 | 1, l, r)) % p;
}
signed main() {
    int n, m;
    cin >> n >> m >> p;
    for (int i = 1; i <= n; i++) cin >> a[i];
    build(1, 1, n);
    for (int i = 0; i < m; i++) {
        int op, l, r, k;
        cin >> op >> l >> r;
        if (op == 1) {
            cin >> k;
            modify(1, l, r, 0, k);
        } else if (op == 2) {
            cin >> k;
            modify(1, l, r, k, 1);
        } else {
            cout << query_sum(1, l, r) << endl;
        }
    }
    return 0;
}

动态申请空间

#include <iostream>
using namespace std;
const int N = 1e5 + 5;
const int INF = 1e9 + 5;
struct Node {
    int lc, rc, l, r, cnt, del;
}tr[N * 50];
int rt, idx;
void pushup(int u) {
    int lc = tr[u].lc, rc = tr[u].rc;
    tr[u].cnt = tr[lc].cnt + tr[rc].cnt;
}
void update(int u) {
    tr[u].del = 1;
    tr[u].cnt = 0;
}
void pushdown(int u) {
    if (!tr[u].del) return;
    update(tr[u].lc), update(tr[u].rc);
    tr[u].del = 0;
}
void insert(int &u, int l, int r, int x) {
    if (!u) u = ++idx;
    tr[u].l = l, tr[u].r = r;
    if (l == r) {
        tr[u].del = 0;
        tr[u].cnt++;
        return;
    }
    pushdown(u);
    int mid = l + r >> 1;
    if (x <= mid) insert(tr[u].lc, l, mid, x);
    else insert(tr[u].rc, mid + 1, r, x);
    pushup(u);
}
void modify(int u, int l, int r) {
    if (!u || tr[u].del || !tr[u].cnt) return;
    if (tr[u].l > r || tr[u].r < l) return;
    if (tr[u].l >= l && tr[u].r <= r) {
        update(u);
        return;
    }
    pushdown(u);
    modify(tr[u].lc, l, r);
    modify(tr[u].rc, l, r);
    pushup(u);
}
int query_sum(int u, int l, int r) {
    if (!u || tr[u].del || !tr[u].cnt) return 0;
    if (tr[u].l > r || tr[u].r < l) return 0;
    if (tr[u].l >=l && tr[u].r <= r) return tr[u].cnt;
    return query_sum(tr[u].lc, l, r) + query_sum(tr[u].rc, l, r);
}
int query_kth(int u, int l, int r, int k) {
    if (!u || tr[u].del || !tr[u].cnt) return -1;
    if (tr[u].l == tr[u].r) return tr[u].cnt >= k ? tr[u].r : -1;
    pushdown(u);
    int mid = tr[u].l + tr[u].r >> 1;
    int res = 0;
    if (r > mid) {
        res = query_sum(tr[u].rc, l, r);
        if (res >= k) return query_kth(tr[u].rc, l, r, k);
    }
    if (l <= mid) return query_kth(tr[u].lc, l, r, k - res);
    return -1;
}
int main() {
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        int op, l, r, x, k;
        scanf("%d", &op);
        if (op == 1) {
            scanf("%d", &x);
            insert(rt, 1, INF, x);
        } else if (op == 2) {
            scanf("%d%d", &l, &r);
            modify(1, l, r);
        } else {
            scanf("%d%d%d", &l, &r, &k);
            printf("%d\n", query_kth(1, l, r, k));
        }
    }
    return 0;
}

线段树解决树上问题

CF1192 B.Dynamic Diameter

对一棵带权无向树，动态维护树的直径。

#include <iostream>
#include <cstring>
using namespace std;
#define int long long 
const int N = 1e5 + 5;
const int inf = -1e18;
int h[N], e[N * 2], ne[N * 2], w[N * 2], idx = 1;
int d[N], f[N], weight[N], fi[N], la[N], which[N * 2], tot;
struct Node {
    int l, r;
    int val, M, LM, MR, LMR;
    int add; 
}tr[N * 8];
void pushup(int u) {
    Node &rt = tr[u], &lc = tr[u << 1], &rc = tr[u << 1 | 1];
    rt.val = max(lc.val, rc.val);
    rt.M = max(lc.M, rc.M);
    rt.LM = max(max(lc.LM, rc.LM), lc.val + rc.M);
    rt.MR = max(max(lc.MR, rc.MR), lc.M + rc.val);
    rt.LMR = max(max(lc.LMR, rc.LMR), max(lc.LM + rc.val, lc.val + rc.MR)); 
}
void pushdown(int u) {
    if (tr[u].add == 0) return;
    Node &rt = tr[u], &lc = tr[u << 1], &rc = tr[u << 1 | 1];
    lc.val += rt.add; rc.val += rt.add;
    lc.M -= 2 * rt.add; rc.M -= 2 * rt.add;
    lc.LM -= rt.add; rc.LM -= rt.add;
    lc.MR -= rt.add; rc.MR -= rt.add;
    lc.add += rt.add; rc.add += rt.add;
    rt.add = 0;
}
void build(int u, int l, int r) {
    tr[u] = {l, r, -inf, -inf, -inf, -inf, -inf};
    if (l == r) {
        int x = which[r];
        tr[u] = {l, r, d[x], -2 * d[x], -d[x], -d[x]};
    } else {
        int mid = l + r >> 1;
        build(u << 1, l, mid);
        build(u << 1 | 1, mid + 1, r);
        pushup(u);
    }
}
void modify(int u, int l, int r, int k) {
    if (tr[u].l > r || tr[u].r < l) return;
    if (tr[u].l >= l && tr[u].r <= r) {
        tr[u].add += k;
        tr[u].val += k; tr[u].M -= 2 * k;
        tr[u].LM -= k; tr[u].MR -= k;
    } else {
        pushdown(u);
        modify(u << 1, l, r, k);
        modify(u << 1 | 1, l, r, k);
        pushup(u);
    }
}
void adde(int a, int b, int c) {
    e[++idx] = b, w[idx] = c, ne[idx] = h[a], h[a] = idx; 
}
void dfs(int u, int fa, int s) {
    which[fi[u] = ++tot] = u;
    d[u] = s;
    f[u] = fa;
    for (int i = h[u]; i; i = ne[i]) {
        int j = e[i];
        if (j == fa) continue;
        weight[j] = w[i];
        dfs(j, u, s + w[i]);
        which[++tot] = u;
    }
    la[u] = tot;
}
signed main() {
    int n, q, W;
    scanf("%lld%lld%lld", &n, &q, &W);
    for (int i = 1; i < n; i++) {
        int a, b, c;
        scanf("%lld%lld%lld", &a, &b, &c);
        adde(a, b, c), adde(b, a, c);
    }
    dfs(1, 1, 0);
    build(1, 1, tot);
    int res = 0;
    for (int i = 0; i < q; i++) {
        int x, y;
        scanf("%lld%lld", &x, &y);
        x = (x + res) % (n - 1) + 1;
        y = (y + res) % W;
        if(f[e[x << 1]] == e[x << 1 | 1]) x = e[x << 1];
        else x = e[x << 1 | 1];
        modify(1, fi[x], la[x], y - weight[x]);
        weight[x] = y;
        res = tr[1].LMR;
        printf("%lld\n", res); 
    }
    return 0;
}

AcWing256.最大异或和

有历史版本的字典树，可进行区间查询操作

#include <iostream>
#include <cmath>
using namespace std;
const int N = 6e5 + 5, M = N * 24;
int s[N], root[N], tr[M][2], maxid[M], idx;
void insert(int i, int k, int lst, int u) {
    if (k < 0) {
        maxid[u] = i;
        return;
    }
    int v = s[i] >> k & 1;
    if (lst) tr[u][v ^ 1] = tr[lst][v ^ 1];
    tr[u][v] = ++idx;
    insert(i, k - 1, tr[lst][v], tr[u][v]);
    maxid[u] = max(maxid[tr[u][0]], maxid[tr[u][1]]);
}
int query(int u, int C, int L) {
    for (int i = 23; i >= 0; i--) {
        int v = C >> i & 1;
        if (maxid[tr[u][v ^ 1]] >= L) u = tr[u][v ^ 1];
        else u = tr[u][v];
    }
    return s[maxid[u]] ^ C;
}
int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    maxid[0] = -1;
    root[0] = ++idx;
    insert(0, 23, 0, root[0]);
    for (int i = 1; i <= n; i++) {
        int x;
        scanf("%d", &x);
        s[i] = s[i - 1] ^ x;
        root[i] = ++idx;
        insert(i, 23, root[i - 1], root[i]);
    }
    char op[2];
    int l, r, x;
    while (m--) {
        scanf("%s", op);
        if (*op == 'A') {
            scanf("%d", &x);
            n++;
            s[n] = s[n - 1] ^ x;
            root[n] = ++idx;
            insert(n, 23, root[n - 1], root[n]);    
        } else {  
            scanf("%d%d%d", &l, &r, &x);
            printf("%d\n", query(root[r - 1], s[n] ^ x, l - 1));
        }    
    }
    return 0;
}

可持久化线段树（主席树）

AcWing255.第K小数

可查询区间第k小
一开始就把树建出来的写法

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
const int N = 1e5 + 5;
struct Node {
    int l, r, cnt;
}tr[N * 4 + N * 17];
int root[N], idx;
int n, m;
vector<int> v;
int a[N];
int build(int l, int r) {
    int u = ++idx;
    if (l == r) return u;
    int mid = l + r >> 1;
    tr[u].l = build(l, mid), tr[u].r = build(mid + 1, r);
    return u; 
}
int insert(int lst, int l, int r, int x) {
    int u = ++idx;
    tr[u] = tr[lst];
    if (l == r) {
        tr[u].cnt++;
        return u;
    }
    int mid = l + r >> 1;
    if (x <= mid) tr[u].l = insert(tr[lst].l, l, mid, x);
    else tr[u].r = insert(tr[lst].r, mid + 1, r, x);
    tr[u].cnt = tr[tr[u].l].cnt + tr[tr[u].r].cnt;
    return u;
}
int query(int u, int lst, int l, int r, int k) {
    if (l == r) return r;
    int cnt = tr[tr[u].l].cnt - tr[tr[lst].l].cnt;
    int mid = l + r >> 1;
    if (k <= cnt) return query(tr[u].l, tr[lst].l, l, mid, k);
    else return query(tr[u].r, tr[lst].r, mid + 1, r, k - cnt);
}
int find(int x) {
    return lower_bound(v.begin(), v.end(), x) - v.begin();
}
int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i++) {
        scanf("%d", &a[i]);
        v.push_back(a[i]);
    }
    sort(v.begin(), v.end());
    v.erase(unique(v.begin(), v.end()), v.end());
    root[0] = build(0, v.size() - 1); 
    for (int i = 1; i <= n; i++) {
        root[i] = insert(root[i - 1], 0, v.size() - 1, find(a[i]));
    }
    while (m--) {
        int l, r, k;
        scanf("%d%d%d", &l, &r, &k);
        printf("%d\n", v[query(root[r], root[l - 1], 0, v.size() - 1, k)]);
    }
    return 0;
}

2021ICPC昆明M.Stone Games

可查询区间比一个数小的数的和
动态开点的写法

#include <iostream>
using namespace std;
#define int long long 
const int N = 1e6 + 5;
const int MAX = 1e9;
struct Node {
    int l, r, cnt;
}tr[N * 50];
int root[N], idx;
int insert(int lst, int l, int r, int x) {
    int u = ++idx;
    tr[u] = tr[lst];
    tr[u].cnt += x; 
    if (l == r) return u;
    int mid = l + r >> 1;
    if (x <= mid) tr[u].l = insert(tr[lst].l, l, mid, x);
    else tr[u].r = insert(tr[lst].r, mid + 1, r, x);
    return u;
}
int query(int u, int lst, int l, int r, int x) {
    if (l == r) return tr[u].cnt - tr[lst].cnt;
    int mid = l + r >> 1;
    if (x <= mid) return query(tr[u].l, tr[lst].l, l, mid, x);
    else return query(tr[u].r, tr[lst].r, mid + 1, r, x) + tr[tr[u].l].cnt - tr[tr[lst].l].cnt;
}
signed main() {
    int n, Q;
    scanf("%lld%lld", &n, &Q);
    for (int i = 1; i <= n; i++) {
        int x;
        scanf("%lld", &x);
        root[i] = insert(root[i - 1], 1, MAX, x);
    }
    int ans = 0;
    while (Q--) {
        int l, r;
        scanf("%lld%lld", &l, &r);
        l = (l + ans) % n + 1, r = (r + ans) % n + 1;
        if (l > r) swap(l, r);
        int x = 1;
        while (1) {
            int s = query(root[r], root[l - 1], 1, MAX, x);
            if (s == x - 1) break;
            x = s + 1;
        }
        printf("%lld\n", ans = x);
    }
    return 0;
}

平衡树

Splay

P3369 【模板】普通平衡树

插入x数
删除x数(若有多个相同的数，因只删除一个)
查询x数的排名(排名定义为比当前数小的数的个数+1)
查询排名为x的数
求x的前驱(前驱定义为小于 x，且最大的数)
求x的后继(后继定义为大于 x，且最小的数)

#include <iostream>
using namespace std;
const int N = 1e5 + 5, INF = 0x3f3f3f3f;
struct Node {
    int s[2], val, fa, cnt, sz;
    void init(int v, int f) {
        s[0] = s[1] = 0;
        val = v, fa = f, cnt = sz = 1;
    }
}tr[N];
int root, idx;
void pushup(int x) {
    tr[x].sz = tr[tr[x].s[0]].sz + tr[tr[x].s[1]].sz + tr[x].cnt;
}
int get(int x) {
    return x == tr[tr[x].fa].s[1];
}
void Rotate(int x) {
    int y = tr[x].fa, z = tr[y].fa;
    int k = get(x), sonx = tr[x].s[k ^ 1];
    if (z) tr[z].s[get(y)] = x;
    tr[y].s[k] = sonx;
    if (sonx) tr[sonx].fa = y;
    tr[x].s[k ^ 1] = y, tr[y].fa = x, tr[x].fa = z;
    pushup(y), pushup(x);
}
void splay(int x, int k = 0) {
    for (int y; y = tr[x].fa, y != k; Rotate(x)) {
        if (tr[y].fa != k) Rotate(get(x) == get(y) ? y : x);
    }
    if (!k) root = x;
}
void ins(int k) {
    int x = root, y = tr[x].fa;
    while (x) {
        if (tr[x].val == k) { tr[x].cnt++; splay(x); return; }
        y = x;
        x = tr[x].s[tr[x].val < k];
    }
    tr[++idx].init(k, y);
    if (y) tr[y].s[tr[y].val < k] = idx;
    splay(idx);
}
int rk(int k) {
    int res = 0, x = root;
    while (x) {
        if (tr[x].val > k) x = tr[x].s[0];
        else {
            res += tr[tr[x].s[0]].sz;
            if (tr[x].val == k) {
                splay(x);
                return res + 1;
            }
            res += tr[x].cnt;
            x = tr[x].s[1];
        }
    }
    return res + 1;
}
int kth(int k) {
    int x = root;
    while (x) {
        if (tr[x].s[0] && tr[tr[x].s[0]].sz >= k) x = tr[x].s[0];
        else {
            k -= tr[tr[x].s[0]].sz + tr[x].cnt;
            if (k <= 0) {
                splay(x);
                return tr[x].val;
            }
            x = tr[x].s[1];
        }
    }
    return INF;
}
void del(int k) {
    rk(k);
    if (tr[root].cnt > 1) {
        tr[root].cnt--; tr[root].sz--;
        return;
    }
    if (!tr[root].s[0] || !tr[root].s[1]) {
        root = tr[root].s[0] | tr[root].s[1];
        if (root) tr[root].fa = 0;
        return;
    }
    int x = tr[root].s[0];
    while (tr[x].s[1]) x = tr[x].s[1];
    splay(x);
    tr[x].s[1] = tr[tr[x].s[1]].s[1];
    tr[tr[x].s[1]].fa = x;
    pushup(x);
}
int pre(int k) {
    ins(k);
    int x = tr[root].s[0];
    if (x == 0) return -INF;
    while (tr[x].s[1]) x = tr[x].s[1];
    splay(x);
    del(k);
    return tr[x].val;
}
int nxt(int k) {
    ins(k);
    int x = tr[root].s[1];
    if (x == 0) return INF;
    while (tr[x].s[0]) x = tr[x].s[0];
    splay(x);
    del(k);
    return tr[x].val;
}
int main() {
    int n;
    scanf("%d", &n);
    for (int i = 0, opt, x; i < n; i++) {
        scanf("%d%d", &opt, &x);
        if (opt == 1) ins(x);
        else if (opt == 2) del(x);
        else if (opt == 3) printf("%d\n", rk(x));
        else if (opt == 4) printf("%d\n", kth(x));
        else if (opt == 5) printf("%d\n", pre(x));
        else printf("%d\n", nxt(x));
    }
    return 0;
}
